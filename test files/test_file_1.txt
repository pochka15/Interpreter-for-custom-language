add_one (element int) int {
     ret element + 1
}

print_at_even_pos (elements IntList) None {
    var i int = 0
    for val in elements {
        let is_even bool = i % 2 == 0
        print(if is_even { str(val) } else { "" })
        i = i + 1
    }
}
